Создайте таблицу logs типа Archive. 
Пусть при каждом создании записи в таблицах users, catalogs и products в таблицу logs 
помещается время и дата создания записи, название таблицы, идентификатор первичного ключа и содержимое поля name.

DROP TABLE IF EXISTS logs;
CREATE TABLE logs (
id SERIAL,
tables_name VARCHAR(255) COMMENT 'Имена таблиц',
tables_id INT  COMMENT 'id',
name VARCHAR(255) COMMENT 'Имена в таблицах',
created_at DATETIME COMMENT 'время создания'
) COMMENT = 'логи' ENGINE = Archive;

DESCRIBE logs;

DELIMITER //

CREATE TRIGGER tr_logs_users AFTER INSERT ON users
FOR EACH ROW BEGIN
  INSERT INTO logs (tables_name,  tables_id, name, created_at) 
  VALUES ('users', NEW.id, NEW.name, NEW.created_at);
END//

CREATE TRIGGER tr_logs_catalogs AFTER INSERT ON catalogs
FOR EACH ROW BEGIN
  INSERT INTO logs (tables_name,  tables_id, name, created_at) 
  VALUES ('catalogs', NEW.id, NEW.name, NOW());
END//

CREATE TRIGGER tr_logs_products AFTER INSERT ON products
FOR EACH ROW BEGIN
  INSERT INTO logs (tables_name,  tables_id, name, created_at) 
  VALUES ('products', NEW.id, NEW.name, NEW.created_at);
END//
 
DELIMITER ;
