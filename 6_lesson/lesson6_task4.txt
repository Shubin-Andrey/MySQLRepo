 4. Подсчитать количество лайков которые получили 10 самых молодых пользователей.

SELECT (SELECT COUNT(target_id) FROM likes WHERE likes.target_id = profiles.user_id GROUP BY target_id) AS counter, 
TIMESTAMPDIFF(YEAR, birthday, NOW()) AS age FROM profiles ORDER BY age LIMIT 10 ;
